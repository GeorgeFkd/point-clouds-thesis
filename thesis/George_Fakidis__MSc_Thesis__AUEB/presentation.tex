\section{Background Information}
\begin{frame}{Background Information}
	\begin{itemize}
		\item \textbf{Point clouds} are sets of 3D points $(x, y, z)$, optionally enriched with color and reflectance data, captured by devices such as LiDAR cameras and stereo-depth cameras. We use the Intel RealSense D435I.
		\item \textbf{Object detection} classifies objects and finds their spatial bounds, a rectangle in 2D and a 3D bounding box in point clouds.
		\item \textbf{Face detection} is a specific case of object detection, not to be confused with face recognition which matches faces against a database of known identities.
		\item \textbf{Semantic segmentation} assigns a class label to every point in the cloud, rather than predicting bounding boxes around objects.
		\item \textbf{Privacy} in images is most commonly achieved using blurring and pixelation. Such simple methods are not enough to remove identifiable information \cite{Reid_Samangooei_Chen_Nixon_Ross_2013} and we will see more effective approaches.
	\end{itemize}

	\vfill
	{\small \textit{Note: A bug was found in the RealSense SDK's point-cloud file export during this work and \href{https://github.com/realsenseai/librealsense/pull/14711}{I contributed it upstream as a fix}.}}
\end{frame}



\section{3D Object detection \& Semantic Segmentation models}
\begin{frame}{3D Object Detection \& Semantic Segmentation Models}
	\begin{itemize}
		\item \textbf{PointNet} \cite{qi_pointnet_2017} --- foundational work, operates directly on raw point clouds using MLPs and max-pooling to achieve permutation invariance. Extended by \textbf{PointNet++} with hierarchical local feature extraction.
		\item \textbf{PointPillars} \cite{Lang_Vora_Caesar_Zhou_Yang_Beijbom_2019} --- slices the point cloud into vertical pillars instead of full voxelization, achieving 16ms detection. Used in our experiments.
		\item \textbf{Complex-YOLO} \cite{Simony_Milzy_Amendey_Gross_2018} --- projects the point cloud to a Bird's Eye View and feeds it into a adapted YOLO architecture, achieving 50fps.
		\item \textbf{RandLA-Net} \cite{Hu_Yang_Xie_Rosa_Guo_Wang_Trigoni_Markham_2020} and \textbf{KP-FCNN} \cite{Thomas_Qi_Deschaud_Marcotegui_Goulette_Guibas_2019} --- semantic segmentation models used in our experiments with pretrained weights.
		\item Other notable architectures include Transformer-based models such as \textbf{PointFormer} \cite{pan_3d_2021}, graph-based models such as \textbf{Point-GNN} \cite{shi_point-gnn_2020}, and two-stage detectors such as \textbf{PointRCNN} \cite{shi_pointrcnn_2019}.
	\end{itemize}
\end{frame}

\section{Software Demonstration}

	\begin{enumerate}
		\item Reads depth and color frames at each iteration
		\item Face and people detection runs on each frame
		\item Detection results passed to the rendering engine
		\item Blurring effects are applied and the point cloud is rendered

\section{Discussion}
\begin{frame}{Discussion}
	\begin{itemize}
		\item Face detection != Face recognition, lots of work in 3D face recognition, less on 3D face detection.
		\item Traffic settings and outdoor scenes receive most of the research attention.
		\item Implementing operations on point clouds is more complex.
		\item Point cloud and 3D object detection ecosystem in python is not production ready.
		\item There is a research/software gap for an end-to-end pipeline that anonymizes of 3D videos.
	\end{itemize}
\end{frame}
